<script>
  let inventory = JSON.parse(localStorage.getItem('farmInv_v2')) || [
    { id: 1, item: "Diesel", category: "Fuel/Gas", stock: 0, needed: 20, notes: "", bought: false },
    { id: 2, item: "Dish Soap", category: "Cleaning", stock: 1, needed: 2, notes: "", bought: false }
  ];

  function save() {
    localStorage.setItem('farmInv_v2', JSON.stringify(inventory));
    document.getElementById('status').innerText = "Last saved: " + new Date().toLocaleTimeString();
  }

  function renderTable() {
    const tbody = document.getElementById("tableBody");
    const search = document.getElementById("searchInput").value.toLowerCase();
    tbody.innerHTML = "";

    inventory.forEach((item, idx) => {
      if (item.item.toLowerCase().includes(search)) {
        const tr = document.createElement("tr");
        tr.innerHTML = `
          <td contenteditable="true" onblur="updateField(${idx}, 'item', this.innerText)">${item.item}</td>
          <td>${item.category}</td>
          <td contenteditable="true" class="${item.stock === 0 && item.needed > 0 ? 'low-stock' : ''}" onblur="updateField(${idx}, 'stock', this.innerText, true)">${item.stock}</td>
          <td contenteditable="true" onblur="updateField(${idx}, 'needed', this.innerText, true)">${item.needed}</td>
          <td><button class="btn-delete" onclick="deleteItem(${idx})">Delete</button></td>
        `;
        tbody.appendChild(tr);
      }
    });
  }

  function updateField(idx, field, val, isNum) {
    inventory[idx][field] = isNum ? (parseInt(val) || 0) : val;
    save();
    if(isNum) renderTable();
  }

  function addItem() {
    const name = document.getElementById("newItem").value;
    if(!name) return;
    inventory.push({
      id: Date.now(),
      item: name,
      category: document.getElementById("newCategory").value,
      stock: parseInt(document.getElementById("newStock").value) || 0,
      needed: parseInt(document.getElementById("newNeeded").value) || 0,
      bought: false
    });
    document.getElementById("newItem").value = "";
    save(); renderTable();
  }

  function toggleMode(isShopping) {
    document.getElementById("mainView").classList.toggle("hidden", isShopping);
    document.getElementById("shoppingView").classList.toggle("hidden", !isShopping);
    if(isShopping) renderShoppingList();
  }

  function renderShoppingList() {
    const listDiv = document.getElementById("shoppingList");
    listDiv.innerHTML = "";
    const needs = inventory.filter(i => i.needed > 0);

    if(needs.length === 0) {
      listDiv.innerHTML = "<p>No items marked as 'Needed'. Add some first!</p>";
      return;
    }

    needs.forEach(item => {
      const div = document.createElement("div");
      div.className = `shopping-item ${item.bought ? 'checked' : ''}`;
      div.onclick = () => {
        item.bought = !item.bought;
        renderShoppingList();
      };
      div.innerHTML = `
        <input type="checkbox" ${item.bought ? 'checked' : ''}>
        <span>${item.item}</span>
        <span class="qty-badge">Buy: ${item.needed}</span>
      `;
      listDiv.appendChild(div);
    });
  }

  function finishTrip() {
    if(!confirm("Ready to update stock? Items checked will move to Current Stock.")) return;
    
    inventory.forEach(item => {
      if(item.bought && item.needed > 0) {
        item.stock += item.needed;
        item.needed = 0;
        item.bought = false; // Reset for next time
      }
    });
    
    save();
    toggleMode(false);
    renderTable();
  }

  function deleteItem(idx) {
    if(confirm("Remove item?")) { inventory.splice(idx,1); save(); renderTable(); }
  }

  function exportToCSV() {
    let csv = "Item,Category,Stock,Needed\n";
    inventory.forEach(i => csv += `"${i.item}","${i.category}",${i.stock},${i.needed}\n`);
    const blob = new Blob([csv], { type: 'text/csv' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = 'farm_inventory.csv';
    a.click();
  }

  renderTable();
</script>
